In your HTML, add a script like this:
<script>
  let deferredPrompt;

  window.addEventListener('beforeinstallprompt', (e) => {
    // Prevent automatic prompt
    e.preventDefault();
    deferredPrompt = e;

    // Show your custom install button or popup
    document.getElementById("installPrompt").style.display = "block";
  });

  function installApp() {
    if (deferredPrompt) {
      deferredPrompt.prompt();
      deferredPrompt.userChoice.then((choiceResult) => {
        if (choiceResult.outcome === 'accepted') {
          console.log('User accepted the install prompt');
        } else {
          console.log('User dismissed the install prompt');
        }
        deferredPrompt = null;
        document.getElementById("installPrompt").style.display = "none";
      });
    }
  }
</script>





Put this somewhere in your HTML body

<div id="installPrompt" style="display:none; position:fixed; bottom:20px; left:20px; background:#0d47a1; color:#fff; padding:15px; border-radius:10px; box-shadow:0 0 10px rgba(0,0,0,0.3); z-index:9999;">
  <p style="margin:0 0 10px;">Install Xander Media on your device?</p>
  <button onclick="installApp()" style="padding:8px 12px; background:#fff; color:#0d47a1; border:none; border-radius:5px;">Install</button>
